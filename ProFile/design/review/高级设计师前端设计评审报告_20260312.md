  assignee?: string;
  deadline: string;

  updatedAt: string;

interface Project {
  id: string;
  name: string;
  description: string;
  status: 'active' | 'completed' | 'archived';
  priority: 'P0' | 'P1' | 'P2' | 'P3';
  manager?: string;
  startDate: string;
  endDate: string;
  tasks: Task[];
}
