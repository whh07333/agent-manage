## 4. 错误处理规范

### 4.1 错误类型定义
```typescript
// 自定义错误类
class AppError extends Error {
  constructor(
    public readonly message: string,
    public readonly code: string,
    public readonly statusCode: number = 500
  ) {
    super(message);
    this.name = 'AppError';
  }
}

class ValidationError extends AppError {
  constructor(message: string, public readonly errors: FieldError[]) {
    super(message, 'VALIDATION_ERROR', 400);
    this.name = 'ValidationError';
  }
}

class NotFoundError extends AppError {
  constructor(resource: string, id: string) {
    super(`${resource} with id ${id} not found`, 'NOT_FOUND', 404);
    this.name = 'NotFoundError';
  }
}
```

### 4.2 错误处理中间件
```typescript
// 统一的错误处理中间件
export const errorHandler = (
  err: Error,
  req: Request,
  res: Response,
  next: NextFunction
) => {
  if (err instanceof AppError) {
    return res.status(err.statusCode).json({
      code: err.code,
      msg: err.message,
      data: null,
      trace_id: req.headers['x-request-id']
    });
  }

  // 未知错误
  console.error('Unexpected error:', err);
  return res.status(500).json({
    code: 'INTERNAL_ERROR',
    msg: 'Internal server error',
    data: null,
    trace_id: req.headers['x-request-id']
  });
};
```

## 8. 安全规范

### 8.1 输入验证
```typescript
// 使用类验证器进行输入验证
class CreateProjectValidator {
  static validate(data: any): ValidationResult {
    const errors: FieldError[] = [];

    if (!data.name || data.name.trim().length === 0) {
      errors.push({ field: 'name', message: '项目名称不能为空' });
    }

    if (data.name && data.name.length > 255) {
      errors.push({ field: 'name', message: '项目名称不能超过255个字符' });
    }

    if (data.priority && !['P0', 'P1', 'P2', 'P3'].includes(data.priority)) {
      errors.push({ field: 'priority', message: '优先级值无效' });
    }

    return {
      isValid: errors.length === 0,
      errors
    };
  }
}
```

// 使用参数化查询，禁止字符串拼接
