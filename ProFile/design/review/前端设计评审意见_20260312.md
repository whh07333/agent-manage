```javascript
// 使用ECharts实现燃尽图
const BurnDownChart = ({ data }) => {
  const chartRef = useRef(null);
  
  useEffect(() => {
    const chart = echarts.init(chartRef.current);
    
    const option = {
      tooltip: {
        trigger: 'axis',
        formatter: '{b}: {c}天'
      },
      xAxis: {
        type: 'category',
        data: data.dates
      },
      yAxis: {
        type: 'value',
        name: '剩余任务数'
      },
      series: [
        {
          data: data.remaining,
          type: 'line',
          smooth: true,
          itemStyle: {
            color: '#1890ff'
          }
        }
      ]
    };
    
    chart.setOption(option);
    
    return () => chart.dispose();
  }, [data]);
  
  return <div ref={chartRef} style={{ height: '400px' }} />;
};
```
